    /* Function to generate the week cards for each month */
    function generateCalendar() {
      courseWeeks.forEach((week) => {
        const container = document.getElementById(`${week.month}-weeks`);
        if (container) {
          // Create a card for the week
          const card = document.createElement('div');
          card.className = 'week-card';
          card.onclick = () => openModal(week); // open the modal with data

          const header = document.createElement('h3');
          header.textContent = `Week ${week.weekNumber}: ${week.dateRange}`;
          card.appendChild(header);

          const topicP = document.createElement('p');
          topicP.textContent = `Topic: ${week.topic}`;
          card.appendChild(topicP);

          container.appendChild(card);
        }
      });
    }

    /* Modal handling */

    const closeModalBtn = document.getElementById('closeModal');

    function openModal(week) {
      modalTitle.textContent = `Week ${week.weekNumber} â€“ ${week.dateRange}`;

      modalAssessments.innerHTML = `<strong>Assessments:</strong> ${week.assessments}`;
      modalOverlay.classList.add('active');
    }

    function closeModal() {
      modalOverlay.classList.remove('active');

    closeModalBtn.addEventListener('click', closeModal);

        tabs.forEach((t) => t.classList.remove('active'));
        // Hide all calendar containers
        calendars.forEach((c) => c.classList.remove('active'));

        // Activate current tab and corresponding calendar
        tab.classList.add('active');
        const monthId = tab.getAttribute('data-month');
        document.getElementById(monthId).classList.add('active');
      });
